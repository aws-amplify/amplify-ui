Add default dark mode theme override for React

```jsx
  const theme: Theme = {
    name: 'my-theme',
    overrides: [defaultDarkModeOverride],
  };
  
  // ...
  <AmplifyProvider theme={theme} colorMode="system">
  
  // or
  <AmplifyProvider theme={theme} colorMode={colorMode}>
```
