import { Tabs, TabItem } from '@aws-amplify/ui-react';

For Vue 3 install `@aws-amplify/ui-vue@next` with [npm](https://www.npmjs.com/) or [yarn](https://yarnpkg.com/):

<Tabs>
<TabItem title="npm">

```shell
npm install aws-amplify @aws-amplify/ui-vue@next
```

</TabItem>
<TabItem title="yarn">

```shell
yarn add @aws-amplify/ui-vue@next
```

</TabItem>
</Tabs>

**For Vue 2 please follow the installation instructions listed [here.](https://docs.amplify.aws/ui/q/framework/vue/)**
