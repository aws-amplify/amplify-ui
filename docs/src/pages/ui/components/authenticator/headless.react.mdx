Use the `useAuth` hook to get access to the state machine and customize your UI:

```js{1,4-6,10}
import { useAuth } from "@aws-amplify/ui-react";

function App() {
  const [state, send] = useAuth();

  if (state.matches("authenticated")) {
    return (
      <>
        <h1>Welcome, {state.context.user.username}</h1>
        <button onClick={() => send("SIGN_OUT")}>Sign out</button>
      </>
    );
  }

  ...
}
```
