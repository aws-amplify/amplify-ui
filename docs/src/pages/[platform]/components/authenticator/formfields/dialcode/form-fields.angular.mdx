_app.component.ts_

```js
import { Component } from '@angular/core';
import { Amplify } from 'aws-amplify';

import awsExports from './aws-exports';

@Component({
  selector: 'app',
  templateUrl: 'app.component.html',
})
export class AppComponent {
  constructor() {
    Amplify.configure(awsExports);
  }
  public formFields = {
    signIn: {
      username: {
        dialCode: "+227"
      },
    },
    signUp: {
      phone_Number: {
        dialCode: "+227"
      },
    },
  };
}
```
